name: Convert Google Sheet to JSON

on:
  schedule:
    - cron: '0 * * * *'  # Chạy mỗi giờ (bạn có thể điều chỉnh cron theo nhu cầu)
  workflow_dispatch:       # Cho phép chạy thủ công

jobs:
  convert:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout repository
        uses: actions/checkout@v2

      - name: Setup Node.js
        uses: actions/setup-node@v2
        with:
          node-version: '14'

      - name: Install dependencies
        run: npm install xlsx

      - name: Download Excel from Google Sheets and convert to JSON
        run: |
          echo "Downloading Excel from Google Sheets..."
          wget -O book1.xlsx "https://docs.google.com/spreadsheets/d/1mMiVqnAE_y5uEH8MS7zk5BAF6w1P50Br/export?format=xlsx"
          echo "Converting Excel to JSON..."
          node -e "const xlsx = require('xlsx'); const fs = require('fs'); const workbook = xlsx.readFile('book1.xlsx'); const sheetName = workbook.SheetNames[0]; const jsonData = xlsx.utils.sheet_to_json(workbook.Sheets[sheetName]); fs.writeFileSync('data.json', JSON.stringify(jsonData, null, 2)); console.log('Conversion done.');"

      - name: Commit and push changes
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
        run: |
          git config --local user.email "action@github.com"
          git config --local user.name "GitHub Action"
          git add data.json
          # Kiểm tra nếu có thay đổi, commit và push
          git diff --cached --exit-code || git commit -m "Auto-update data.json from Google Sheets"
          git push
